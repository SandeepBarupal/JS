Example

function wait(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

async function loadAll() {
  const p1 = wait(500).then(() => 'A');
  const p2 = wait(1000).then(() => 'B');
  const results = await Promise.all([p1, p2]);
  console.log(results);
}
loadAll();
