Example

const Counter = (function() {
  let count = 0;
  return {
    increment() { count++; },
    getCount() { return count; }
  };
})();
Counter.increment();
console.log(Counter.getCount());

function createUser(name, role) {
  return {
    name,
    role,
    describe() { console.log(`${name} is a ${role}`); }
  };
}
const user = createUser('Asha', 'Teacher');
user.describe();
