Practice Tasks

  2. Function composition ka example banao.

Example

const add = a => b => a + b;
const add5 = add(5);
console.log(add5(10));

const multiply = a => b => a * b;
const times2 = multiply(2);
console.log(times2(7));

const compose = (f, g) => x => f(g(x));
const square = x => x * x;
const addOne = x => x + 1;
const squareAfterAdd = compose(square, addOne);
console.log(squareAfterAdd(4));
